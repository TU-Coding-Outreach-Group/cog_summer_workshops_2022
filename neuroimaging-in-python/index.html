# Converting fMRI stimuli timing to TRs
# this script will loop through a list of subject numbers and the runs

subjects = ['049','050','052','053'] #subject numbers the code will loop through
num_runs = ['1','2','3','4'] #number of functional runs 

TR = 2 #make sure this is correct!!!
sampling_freq = 1./TR #keep the same

#directories

os.chdir(data_dir)

for x, subj in enumerate(subjects):
    sub_runs = Design_Matrix(sampling_freq = sampling_freq)

        dm['tour'] = pd.to_numeric(dm['tour'],errors='coerce')
        dm['room'] = pd.to_numeric(dm['room'],errors='coerce')
        dm['new'] = pd.to_numeric(dm['new'],errors='coerce')

        dm['fix'] = pd.to_numeric(dm['fix'],errors='coerce')
        
        dm.to_csv(data_dir + '/onsets_files/dm_'+str(subj)+'_run-'+str(value)+'.csv')
        f,a = plt.subplots(figsize=(20,3))

        plt.savefig(data_dir+'/onsets_files/dm_'+str(subj)+'_run-'+str(value)+'.png')

        # 2) Convolve them with the hrf
        dm_conv = dm.convolve()

        # 2) Load in covariates for this run
        covariatesFile = os.path.join(confounds_dir,"sub-"+str(subj)+"_ses-001_task-objects_run-"+str(value)+"_desc-confounds_timeseries.tsv") #loop through each individual run
        cov = pd.read_csv(covariatesFile,sep = '\t')
        cov = Design_Matrix(cov[['trans_x','trans_y','trans_z','rot_x','rot_y','rot_z']], sampling_freq = sampling_freq)

        # 3) In the covariates, fill any NaNs with 0, add intercept and linear trends and dct basis functions
        cov = cov.fillna(0)

        # Retain a list of nuisance covariates (e.g. motion and spikes) which we'll also want to also keep separate for each run
        cov_columns = cov.columns
        cov = cov.add_poly(1).add_dct_basis() #can remove polynomial nuisances variables if you want low frequency signal

        plt.savefig(data_dir+'/onsets_files/covariates_'+str(subj)+'_run-'+str(value)+'.png')

        # 4) Join the onsets and covariates together
        full = dm_conv.append(cov,axis=1)

        # 5) Append it to the master Design Matrix keeping things separated by run
        sub_runs = sub_runs.append(full,axis=0,unique_cols=cov.columns)

subjects =['049','050','052','053']
num_runs = ['1','2','3','4']

data_dir = '/Users/tuk98863/Desktop/workshop/derivatives' #change to your actual path
bids_dir = '/Users/tuk98863/Desktop/workshop/'
os.chdir(bids_dir)

all_runs_dir = bids_dir 

subjs_scan_info = pd.read_csv(bids_dir+'derivatives/mriqc_summary_poor.csv') # path to csv file with information on data quality for each subject and each run

subjs_list = list(subjs_scan_info['subject'].unique()) #creates a list of unique subject IDs

for x, subj in enumerate(subjects):
        print('Starting 1st-level analysis for '+subj)
        
        #goes thorugh the mriqc csv and creates a list of subject IDs and run numbers
        func_runs = list('derivatives/sub-'+subj+'/ses-001/func/'+subjs_scan_info[subjs_scan_info['subject']=='sub-'+subj]['run'].str[:-5]+'_space-MNI152NLin2009cAsym_desc-preproc_bold.nii.gz')
        func_runs.sort()
        
        func_run_nums =  [s.lstrip('0') for s in subjs_scan_info[subjs_scan_info['subject']==subj]['run'].str[-8:-5]]
        func_run_nums = [int(i) for i in func_run_nums]
        print('Number of functional runs for '+subj+': '+str(len(func_runs)))


        # Load the brain data

        # Grab subject's MNI as a mask to keep analysis in MNI space
        subj_MNI = "derivatives/sub-"+subj+"/ses-001/func/sub-"+subj+"_ses-001_task-objects_run-1_space-MNI152NLin2009cAsym_desc-brain_mask.nii.gz"

        print("Loading brain data...")
        all_cat_run_data = Brain_Data(func_runs, mask=subj_MNI)
        
        # smooth
        fwhm = 6
        all_cat_run_data = all_cat_run_data.smooth(fwhm=fwhm)
        
        all_cat_run_data.write(data_dir+'/sub-'+subj+'/sub-'+subj+'_betas_denoised_smoothed6mm_preprocessed_fMRI_bold.nii.gz')

        # Design Matrix

        ## Add design matrices to brain data objects
        all_cat_run_data.X = pd.read_csv(bids_dir+'derivatives/onsets_files/all_runs_'+subj+'.csv', index_col=0)


        ## Estimate model for all voxels
        stats_all_runs = all_cat_run_data.regress()
  
        #create directory to store beta maps
        beta_dir = data_dir + '/beta_maps_6mm'
        if not os.path.exists(beta_dir):
                os.makedirs(beta_dir)
                
        # Save the betas maps for each condition
    
        stats_all_runs['beta'][0].write(os.path.join(beta_dir,'sub-'+subj+'_beta_6mm_fix.nii.gz'))
        stats_all_runs['beta'][1].write(os.path.join(beta_dir,'sub-'+subj+'_beta_6mm_new.nii.gz'))
        stats_all_runs['beta'][2].write(os.path.join(beta_dir,'sub-'+subj+'_beta_6mm_room.nii.gz'))
        stats_all_runs['beta'][3].write(os.path.join(beta_dir,'sub-'+subj+'_beta_6mm_tour.nii.gz'))
  
 
      # Create contrast for all vs rest (fixation)
        c_allVfix = np.zeros(len(stats_all_runs['beta']))
        c_allVfix[list(range(1,4))] = 1/3
        c_allVfix[0] = -1
        allVfix_tmap = stats_all_runs['beta'] * c_allVfix
        allVfix_tmap.write(os.path.join(beta_dir,'sub-'+subj+'tmap_allVfix_6mm.nii'))
        print('finished with ' +subj+ ' smoothed')
